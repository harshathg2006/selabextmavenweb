pipeline {
    agent any

    tools {
        maven 'MAVEN-HOME'
    }

    stages {

        stage('git repo & clean') {
            steps {
                bat "git clone https://github.com/ASHRITHGOUD/maven-se-lab.git"
                bat "mvn clean -f maven-se-lab/pom.xml"
            }
        }

        stage('install') {
            steps {
                bat "mvn install -f maven-se-lab/pom.xml"
            }
        }

        stage('test') {
            steps {
                bat "mvn test -f maven-se-lab/pom.xml"
            }
        }

        stage('package') {
            steps {
                bat 'mvn package -f maven-se-lab/pom.xml'
            }
        }
    }
}
